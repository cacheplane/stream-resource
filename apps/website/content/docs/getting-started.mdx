# Getting Started

Add @cacheplane/langchain to your Angular 20+ application.

## Installation

```bash
npm install @cacheplane/langchain@latest
```

## Setup

In `app.config.ts`, add `provideStreamResource` to your providers:

```typescript
import { provideStreamResource } from '@cacheplane/langchain';

export const appConfig: ApplicationConfig = {
  providers: [
    provideStreamResource({ apiUrl: 'http://localhost:2024' }),
  ],
};
```

## Your first component

```typescript
import { streamResource } from '@cacheplane/langchain';

@Component({
  template: `
    @for (msg of chat.messages(); track $index) {
      <p>{{ msg.content }}</p>
    }
    <button (click)="chat.submit({ messages: [{ role: 'human', content: 'Hello' }] })">
      Send
    </button>
  `,
})
export class ChatComponent {
  chat = streamResource<{ messages: BaseMessage[] }>({
    assistantId: 'chat_agent',
  });
}
```

`streamResource()` MUST be called in an Angular injection context (component field initializer or constructor). Never call it in `ngOnInit`.
