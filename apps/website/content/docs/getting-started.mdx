# Getting Started

Add angular to your Angular 20+ application.

## Installation

```bash
npm install angular@latest
```

## Setup

In `app.config.ts`, add `provideAgent` to your providers:

```typescript
import { provideAgent } from 'angular';

export const appConfig: ApplicationConfig = {
  providers: [
    provideAgent({ apiUrl: 'http://localhost:2024' }),
  ],
};
```

## Your first component

```typescript
import { agent } from 'angular';

@Component({
  template: `
    @for (msg of chat.messages(); track $index) {
      <p>{{ msg.content }}</p>
    }
    <button (click)="chat.submit({ messages: [{ role: 'human', content: 'Hello' }] })">
      Send
    </button>
  `,
})
export class ChatComponent {
  chat = agent<{ messages: BaseMessage[] }>({
    assistantId: 'chat_agent',
  });
}
```

`agent()` MUST be called in an Angular injection context (component field initializer or constructor). Never call it in `ngOnInit`.
